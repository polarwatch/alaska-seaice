190    def compute_extent_km(self, ds: xr.Dataset, area: xr.Dataset)->float:
191        """ Computes sea ice extent in square kilometers.
192
193        Args:
194            ds (xr.Dataset): Binary sea ice concentration dataset.
195            area (xr.Dataset): Grid cell area dataset.
196
197        Returns:
198            float: Sea ice extent in square kilometers.
199        """
200
201        if isinstance(ds, xr.Dataset):
202            if len(ds.data_vars) == 1:
203                ds = ds[list(ds.datavars)[0]]
204            else:
205                raise TypeError(f"Input `ds` is a multi-variable xarray. Provide a DataArray or single variable dataset")
206
207        # Multiply sea ice concentration data by grid cell area to compute extent
208        ice_cell = ds * area / 1e6 # Convert area to square m
209        if isinstance(ice_cell, xr.DataArray):
210            # Sum sea ice extent overxgrid and ygrid, group by time
211            ice_cell.name = 'seaice_extent'  
212            ice_ext = (ice_cell
213                    .groupby("time")
214                    .sum(dim=["xgrid", "ygrid"])) 
215            return ice_ext
216        
217        else:
218            raise TypeError(f"extent values should be in an xarray.DataArray {type(ice_cell)}")

251def clip_data(ds: xr.Dataset, shape:gpd.GeoDataFrame)-> xr.Dataset:
252    """clips data using the shape geometry and returned clipped data
253
254    Args:
255        shape (gpd.GeoDataFrame): crs transformed shape geometry in GeoDataFrame 
256    """
257    
258    # Check if CRS match, if not transform the shape geometry to match data CRS
259
260    if shape.crs != ds.rio.crs:
261        print("Shape CRS does not match data CRS. Performing CRS transformation")
262        shape = shape.to_crs(ds.rio.crs)
263
264    # Perform the clipping operation
265    clipped_ds = ds.rio.clip(shape.geometry.apply(mapping), shape.crs)
266    
267    return clipped_ds
